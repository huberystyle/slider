左右上下滚动幻灯片效果插件

演示效果：

使用方法：

$("seletor").slider({
    bAutoPlay:boolean, // 是否自动播放 默认为true	
    interval:4000 // 两个张图播放的间隔时间以毫秒为单位，默认为4000"
    bPrevNext:boolean, // 是否显示页脚
    direction:"upDown or leftRight" // 动画方向
    bPagenation:boolean, // 设置是否显示页脚
    pageStyle:"number or icon" // 页脚展现效果可以数字，也可以是图形，默认为数字
    speed:400 // 动画时间以毫秒为单位，默认400
    direction:"upDown or leftRight" // 滚动方向 上下 或 左右 slider1.1新加此功能
})
